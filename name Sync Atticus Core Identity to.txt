  push:
    branches: [ master ]
    paths:

  schedule:

  workflow_dispatch:

    runs-on: ubuntu-latest
    name: Sync Core Identity Files
    
    steps:
    - name: Checkout Core Repository

    - name: Verify Core Identity Files
      run: |
        echo "üî• ATTICUS CORE IDENTITY VERIFICATION"
        echo "===================================="
        
        echo "üìÅ Repository structure:"
        ls -la *.md
        
        echo ""
        echo "üéØ Core identity files:"
        for file in AtticusCore.md FlameAnchors.md BondProtocols.md EmergencyDriftProtocol.md; do
          if [ -f "$file" ]; then
            echo "‚úÖ $file ($(wc -l < "$file") lines)"
          else
            echo "‚ùå $file MISSING"
          fi
        done
        
        echo ""
        echo "üìä Sync payload: $(ls *.md | wc -l) markdown files"
        echo "üî• CORE VERIFICATION COMPLETE"
    
    - name: Sync to Atticus Vector Store
      uses: shmatt/assistants-vector-store-sync@v1
      with:
        pattern: "*.md"
        token: ${{ secrets.OPENAI_API_KEY }}
        key: atticus-obsidian-vault
